Using in styled component:

  ```jsx
    import { breaAt, BreakpointSize } from 'BreakPoints'

    const Title = styled.h1`
      font-size: 1rem;

      ${breaAt(BreakpointSize.md)} {
        font-size: 2rem;
      }
    `
  ```
